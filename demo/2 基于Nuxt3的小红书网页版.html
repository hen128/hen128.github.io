# 集成Sequelize

npm i sequelize mysql2
1

``server/db.ts`

import pkg from 'sequelize'
import { initModels } from '~/server/model/init-models'
const { Sequelize } = pkg

const seq = new Sequelize({
    host: '127.0.0.1',
    port: 3306,
    dialect: 'mysql',
    database: 'redbook',
    username: 'root',
    password: '123456'
})

export const DB = initModels(seq)
